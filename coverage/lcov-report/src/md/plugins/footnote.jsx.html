  function footnote_def(state, startLine, endLine, silent) {

      if (state.src.charCodeAt(pos) === 0x20) { return false; }
      if (state.src.charCodeAt(pos) === 0x5D /* ] */) {
        break;
      }
    }
 
    if (pos === start + 2) { return false; } // no empty footnote labels

        insideRef = false,

        token.content  = '';
